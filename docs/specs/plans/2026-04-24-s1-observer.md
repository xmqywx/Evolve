```python
"""Migration 001: add digital_human_id column + dedup table + filter indexes.

Idempotent: re-running on an already-migrated DB is a no-op.
"""
import aiosqlite
from pathlib import Path

AGENT_TABLES = [
    "agent_heartbeats", "agent_deliverables", "agent_discoveries",
    "agent_workflows", "agent_upgrades", "agent_reviews",
]


async def _has_column(db: aiosqlite.Connection, table: str, column: str) -> bool:
    async with db.execute(f"PRAGMA table_info({table})") as cur:
        cols = [row[1] async for row in cur]
    return column in cols


async def _has_table(db: aiosqlite.Connection, name: str) -> bool:
    async with db.execute(
        "SELECT 1 FROM sqlite_master WHERE type='table' AND name=?", (name,)
    ) as cur:
        return await cur.fetchone() is not None


async def run(db_path: str | Path) -> None:
    async with aiosqlite.connect(db_path) as db:
        # 1. Add digital_human_id column (idempotent guard)
        for t in AGENT_TABLES:
            if not await _has_column(db, t, "digital_human_id"):
                await db.execute(
                    f"ALTER TABLE {t} ADD COLUMN digital_human_id TEXT NOT NULL DEFAULT 'executor'"
                )
        # 2. Dedup table
        if not await _has_table(db, "agent_discovery_dedup"):
            await db.execute("""
                CREATE TABLE agent_discovery_dedup (
                    dedup_key TEXT PRIMARY KEY,
                    first_seen_at TIMESTAMP NOT NULL,
                    hit_count INTEGER DEFAULT 1
                )
            """)
        # 3. Filter indexes
        await db.execute(
            "CREATE INDEX IF NOT EXISTS idx_heartbeats_dh   ON agent_heartbeats(digital_human_id, created_at DESC)"
        )
        await db.execute(
            "CREATE INDEX IF NOT EXISTS idx_deliverables_dh ON agent_deliverables(digital_human_id, created_at DESC)"
        )
        await db.execute(
            "CREATE INDEX IF NOT EXISTS idx_discoveries_dh  ON agent_discoveries(digital_human_id, created_at DESC)"
        )
        await db.commit()


if __name__ == "__main__":
    import asyncio, sys
    asyncio.run(run(sys.argv[1]))
```

```python
# ---- Discovery dedup ----
async def insert_dedup(self, dedup_key: str) -> None:
    await self._conn.execute(
        "INSERT OR IGNORE INTO agent_discovery_dedup (dedup_key, first_seen_at) VALUES (?, ?)",
        (dedup_key, now_iso()),
    )
    await self._conn.commit()

async def get_dedup(self, dedup_key: str) -> dict | None:
    async with self._conn.execute(
        "SELECT * FROM agent_discovery_dedup WHERE dedup_key = ?", (dedup_key,)
    ) as cur:
        row = await cur.fetchone()
    if not row:
        return None
    cols = [c[0] for c in cur.description]
    return dict(zip(cols, row))

async def increment_dedup(self, dedup_key: str) -> None:
    await self._conn.execute(
        "UPDATE agent_discovery_dedup SET hit_count = hit_count + 1 WHERE dedup_key = ?",
        (dedup_key,),
    )
    await self._conn.commit()

async def purge_dedup_ttl(self, days: int = 7) -> int:
    cursor = await self._conn.execute(
        "DELETE FROM agent_discovery_dedup WHERE first_seen_at < datetime('now', ?)",
        (f'-{days} days',),
    )
    await self._conn.commit()
    return cursor.rowcount
```

```python
# myagent/digital_humans.py
"""Digital Human registry: config + runtime state + auth tokens."""
from __future__ import annotations
import json
import secrets
import hashlib
from dataclasses import dataclass, field, asdict
from pathlib import Path
from datetime import datetime, timezone


@dataclass
class DHConfig:
    id: str
    persona_dir: str
    cmux_session: str
    provider: str = "codex"
    heartbeat_interval_secs: int = 600
    skill_whitelist: list[str] = field(default_factory=list)
    endpoint_allowlist: list[str] = field(default_factory=list)
    enabled: bool = True


@dataclass
class DHState:
    cmux_session: str | None = None
    started_at: str | None = None
    last_heartbeat_at: str | None = None
    restart_count: int = 0
    last_crash: str | None = None
    auth_token_hash: str | None = None
    enabled: bool = True


def _now() -> str:
    return datetime.now(timezone.utc).isoformat()


def _hash(token: str) -> str:
    return "sha256:" + hashlib.sha256(token.encode()).hexdigest()


class DigitalHumanRegistry:
    def __init__(self, state_root: Path):
        self._configs: dict[str, DHConfig] = {}
        self._states: dict[str, DHState] = {}
        self._tokens: dict[str, str] = {}   # token → dh_id (in-memory only)
        self._state_root = Path(state_root)
        self._state_root.mkdir(parents=True, exist_ok=True)

    def register(self, config: DHConfig) -> None:
        self._configs[config.id] = config
        self._states[config.id] = self._load_state(config.id)

    def list_ids(self) -> list[str]:
        return list(self._configs.keys())

    def get_config(self, dh_id: str) -> DHConfig | None:
        return self._configs.get(dh_id)

    def get_state(self, dh_id: str) -> DHState | None:
        return self._states.get(dh_id)

    def _state_path(self, dh_id: str) -> Path:
        d = self._state_root / dh_id
        d.mkdir(parents=True, exist_ok=True)
        return d / "state.json"

    def _load_state(self, dh_id: str) -> DHState:
        p = self._state_path(dh_id)
        if not p.exists():
            return DHState()
        try:
            return DHState(**json.loads(p.read_text()))
        except Exception:
            return DHState()

    def _save_state(self, dh_id: str) -> None:
        state = self._states[dh_id]
        self._state_path(dh_id).write_text(json.dumps(asdict(state), indent=2))

    def mark_started(self, dh_id: str, cmux_session: str) -> None:
        state = self._states[dh_id]
        state.cmux_session = cmux_session
        state.started_at = _now()
        self._save_state(dh_id)

    def record_heartbeat(self, dh_id: str) -> None:
        self._states[dh_id].last_heartbeat_at = _now()
        self._save_state(dh_id)

    def record_crash(self, dh_id: str, reason: str) -> None:
        state = self._states[dh_id]
        state.restart_count += 1
        state.last_crash = reason
        self._save_state(dh_id)


def issue_token(registry: DigitalHumanRegistry, dh_id: str) -> str:
    # Revoke any prior token for this DH
    invalidate_token(registry, dh_id)
    token = secrets.token_urlsafe(32)
    registry._tokens[token] = dh_id
    state = registry._states[dh_id]
    state.auth_token_hash = _hash(token)
    registry._save_state(dh_id)
    return token


def validate_token(registry: DigitalHumanRegistry, token: str) -> str | None:
    return registry._tokens.get(token)


def invalidate_token(registry: DigitalHumanRegistry, dh_id: str) -> None:
    # Remove all tokens mapping to this DH
    to_del = [t for t, v in registry._tokens.items() if v == dh_id]
    for t in to_del:
        del registry._tokens[t]
    state = registry._states.get(dh_id)
    if state:
        state.auth_token_hash = None
        registry._save_state(dh_id)
```

```python
from fastapi import Header, HTTPException, Request


async def auth_dh(request: Request, authorization: str = Header(None)) -> str:
    """FastAPI dependency: return digital_human_id or raise 401."""
    if not authorization or not authorization.startswith("Bearer "):
        raise HTTPException(401, "missing_dh_token")
    token = authorization.removeprefix("Bearer ").strip()
    reg = request.app.state.dh_registry
    dh_id = reg._tokens.get(token)  # inline — registry intentionally exposes
    if not dh_id:
        raise HTTPException(401, "invalid_dh_token")
    return dh_id


def require_endpoint(name: str):
    """Factory returning a dependency that enforces endpoint allowlist."""
    async def _check(request: Request, dh_id: str = Depends(auth_dh)) -> str:
        reg = request.app.state.dh_registry
        cfg = reg.get_config(dh_id)
        if name not in cfg.endpoint_allowlist:
            raise HTTPException(403, f"role_not_permitted: {dh_id} -> {name}")
        return dh_id
    return _check
```

```python
import hashlib
from datetime import datetime, timezone

def _compute_dedup_key(title: str, category: str) -> str:
    date = datetime.now(timezone.utc).strftime("%Y-%m-%d")
    raw = f"{title.lower().strip()}|{category}|{date}"
    return hashlib.sha256(raw.encode()).hexdigest()[:32]

@app.post("/api/agent/discovery")
async def discovery(req: DiscoveryReq, dh_id: str = Depends(require_endpoint("discovery"))):
    key = _compute_dedup_key(req.title, req.category)
    existing = await db.get_dedup(key)
    if existing:
        await db.increment_dedup(key)
        return {"status": "duplicate_suppressed", "hit_count": existing["hit_count"] + 1}
    await db.insert_dedup(key)
    await db.add_discovery(
        title=req.title, category=req.category, content=req.content,
        actionable=req.actionable, priority=req.priority,
        digital_human_id=dh_id,
```

```bash
#!/usr/bin/env bash
PASS=0; FAIL=0
check() { if [ "$2" = "$3" ]; then echo "PASS: $1"; PASS=$((PASS+1)); else echo "FAIL: $1 — expected $3, got $2"; FAIL=$((FAIL+1)); fi; }

# §9.1 Data contract
for t in agent_heartbeats agent_deliverables agent_discoveries agent_workflows agent_upgrades agent_reviews; do
  n=$(sqlite3 agent.db "SELECT COUNT(*) FROM pragma_table_info('$t') WHERE name='digital_human_id'")
  check "$t has digital_human_id" "$n" "1"
done
test -d persona/executor && check "persona/executor/ exists" "y" "y" || check "persona/executor/" "n" "y"
test -d persona/observer && check "persona/observer/ exists" "y" "y" || check "persona/observer/" "n" "y"
test -f digital_humans/executor/state.json && check "executor state.json" "y" "y"
test -f digital_humans/observer/state.json && check "observer state.json" "y" "y"

# §9.2 Stable ≥7 days
rc=$(jq '.restart_count' digital_humans/observer/state.json)
check "observer restart_count ≤3" "$(( rc <= 3 ? 1 : 0 ))" "1"

# §9.4 Observer output
dc=$(sqlite3 agent.db "SELECT COUNT(*) FROM agent_discoveries WHERE digital_human_id='observer'")
check "observer discovery count ≥50" "$(( dc >= 50 ? 1 : 0 ))" "1"

# §9.5 Dedup
dedup_hits=$(sqlite3 agent.db "SELECT COUNT(*) FROM agent_discovery_dedup WHERE hit_count > 1")
check "dedup hit_count>1 count ≥5" "$(( dedup_hits >= 5 ? 1 : 0 ))" "1"

# §9.6 Role isolation
for t in agent_deliverables agent_workflows agent_upgrades agent_reviews; do
  n=$(sqlite3 agent.db "SELECT COUNT(*) FROM $t WHERE digital_human_id='observer'")
  check "$t zero observer rows" "$n" "0"
done

echo "---"
echo "PASSED: $PASS   FAILED: $FAIL"
[ $FAIL -eq 0 ]
```
